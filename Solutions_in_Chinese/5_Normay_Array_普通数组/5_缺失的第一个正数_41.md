# 一、题目描述
![alt text](<../../Snap_for_Questions/5_Normay_Array_普通数组/截屏2026-06-14 17.08.49.png>)

```python
class Solution:
    def firstMissingPositive(self, nums: List[int]) -> int:
        # 首先拿到数组长度
        n = len(nums)

        # 开始循环，核心思想是1234这样的排序来找最小的正整数
        for i in range(n):
            # 进行该位置的，循环判断进行交换, 满足大于1的正数，再满足两个要进行交换的数 不相等，避免死循环：
            while ( 1 <= nums[i] <= n and nums[i] != nums[nums[i]-1]):
                # 进行交换
                # 变量 存储 为 正确index - 1234中的值，-1即为它本身的index
                correct_index = nums[i]-1
                nums[i], nums[correct_index] = nums[correct_index], nums[i] # 这里重点注意 请使用设的correct_index 变量来，不然会超出时间限制。
        # 每个位置都进行交换的判断和循环之后，进行判断
        for i in range(n):
            # 判断是否为理想的最小正数排序
            if nums[i] != i + 1:
                return i + 1 #直接返回这个不相等的值，就是 我们没有出现的需要找的最小的正整数

        return n + 1 #全部循环完后，发现全正确，则只能为长度的下一个正整数
```

# 八、LeetCode核心代码模式（必须背）

```python
class Solution:
    def firstMissingPositive(self, nums: List[int]) -> int:

        n = len(nums)

        for i in range(n):

            while (
                1 <= nums[i] <= n
                and
                nums[i] != nums[nums[i] - 1]
            ):

                correct_index = nums[i] - 1

                nums[i], nums[correct_index] = \
                    nums[correct_index], nums[i]

        for i in range(n):

            if nums[i] != i + 1:

                return i + 1

        return n + 1
```


## ACM代码

```python
def solve():

    nums = list(map(int, input().split()))

    n = len(nums)

    for i in range(n):

        while (
            1 <= nums[i] <= n
            and
            nums[i] != nums[nums[i] - 1]
        ):

            correct_index = nums[i] - 1

            nums[i], nums[correct_index] = \
                nums[correct_index], nums[i]

    for i in range(n):

        if nums[i] != i + 1:

            print(i + 1)
            return

    print(n + 1)


solve()
```
